REPLACE_OR_CREATE:ROM_f = {
	level = 4
	required_locations_fraction = 0.75
	rule = historical
	
	potential = {
		OR = {
			tag = c:BYZ
			AND = {
				owns = location:rome
				NOT = {
					tag = c:PAP
				}
			}
		}
	}
	
	allow = {
		owns = location:rome
		owns = location:constantinople
		country_rank = country_rank:rank_empire
		is_hegemon = yes
		
		trigger_if = {
			limit = { NOT = { tag = c:BYZ } }
			NOT = { country_exists = c:BYZ }
		}
		OR = {
			religion.group = religion_group:christian
			religion = religion:hellenism_religion
		}
	}
	
	name = ROM
	flag = ROM
	adjective = ROM_ADJ
	tag = ROM
	color = map_ROM
	
	regions = {
		italy_region
		iberia_region
		france_region
		balkan_region
		great_britain_region
		anatolia_region
		south_german_region
		crescent_region
		egypt_region
		maghreb_region
		carpathia_region
		caucasus_region
	}
	form_effect = {
		if = {
			limit = { NOT = { court_language = dialect:classical_latin_dialect } }
			set_court_language = dialect:classical_latin_dialect
		}
		if = {
			limit = { NOT = { capital = location:rome } }
			set_capital = location:rome
		}
	}
}
